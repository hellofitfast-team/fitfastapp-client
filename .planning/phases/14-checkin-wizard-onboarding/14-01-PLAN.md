             handleBack();
           } else {

           }

           if (isRtl) {

             if (isValid && currentStep < STEPS.length) setCurrentStep(currentStep + 1);
           } else {
             handleBack();
           }
         },
         delta: 50,
         preventScrollOnSwipe: false, // Don't block vertical scrolling
         trackTouch: true,
         trackMouse: false,
       });
